\section{Bayes’ Theorem}

\subsection{Weighted Average of Conditional Probabilities}

\textbf{Definition and Setup:}  
Let $A$ and $B$ be events. Event $A$ can be expressed as
\[
A = AB \cup AB^c
\]
since an outcome in $A$ must either be in both $A$ and $B$, or in $A$ and not in $B$. The events $AB$ and $AB^c$ are mutually exclusive.

\textbf{Derivation:}  
By Axiom 3,
\[
\Pr(A) = \Pr(AB) + \Pr(AB^c)
\]
Using the definition of conditional probability,
\[
\Pr(A) = \Pr(A \mid B)\Pr(B) + \Pr(A \mid B^c)\Pr(B^c)
\]
or equivalently,
\[
\Pr(A) = \Pr(A \mid B)\Pr(B) + \Pr(A \mid B^c)[1 - \Pr(B)]
\]

\textbf{Statement:}  
The probability of event $A$ is a weighted average of the conditional probabilities, where the weights are the probabilities of the events on which conditioning is done.

\subsection{Learning by Example — Example 3.1 (Part 1)}

\textbf{Problem Statement:}  
An insurance company classifies people as accident-prone or not accident-prone.
\begin{itemize}
    \item $\Pr(\text{accident within 1 year} \mid \text{accident-prone}) = 0.4$
    \item $\Pr(\text{accident within 1 year} \mid \text{not accident-prone}) = 0.2$
    \item $\Pr(\text{accident-prone}) = 0.3$
\end{itemize}

Find the probability that a new policyholder has an accident within one year.

\textbf{Solution:}  
Let
\begin{itemize}
    \item $A_1$: event that the policyholder has an accident within one year
    \item $A$: event that the policyholder is accident-prone
\end{itemize}

Then,
\[
\Pr(A_1) = \Pr(A_1 \mid A)\Pr(A) + \Pr(A_1 \mid A^c)\Pr(A^c)
\]
\[
= (0.4)(0.3) + (0.2)(0.7) = 0.26
\]

\subsection{Learning by Example — Example 3.1 (Part 2)}

\textbf{Problem Statement:}  
Given that a new policyholder has an accident within one year, find the probability that the policyholder is accident-prone.

\textbf{Solution:}  
The desired probability is
\[
\Pr(A \mid A_1) = \frac{\Pr(A \cap A_1)}{\Pr(A_1)}
\]
Using conditional probability,
\[
\Pr(A \mid A_1) = \frac{\Pr(A)\Pr(A_1 \mid A)}{\Pr(A_1)}
\]
Substituting values,
\[
= \frac{(0.3)(0.4)}{0.26} = \frac{6}{13}
\]

\subsection{Formal Introduction: Law of Total Probability and Bayes’ Formula}

\textbf{Law of Total Probability:}  
Let $B_1, B_2, \ldots, B_n$ be mutually exclusive and exhaustive events. Then,
\[
\Pr(A) = \sum_i \Pr(AB_i)
\]
This result is known as the \textbf{Law of Total Probability}.

\textbf{Bayes’ Formula (Proposition 3.1):}  
Using
\[
\Pr(AB_i) = \Pr(B_i \mid A)\Pr(A)
\]
we obtain
\[
\Pr(B_i \mid A) = \frac{\Pr(A \mid B_i)\Pr(B_i)}{\sum_j \Pr(A \mid B_j)\Pr(B_j)}
\]

\textbf{Terminology:}
\begin{itemize}
    \item $\Pr(B_i)$: \textit{a priori} probability
    \item $\Pr(B_i \mid A)$: \textit{a posteriori} probability
\end{itemize}

\subsection{Learning by Example — Example 3.2 (Cards Problem)}

\textbf{Problem Statement:}  
Three cards:
\begin{itemize}
    \item Card 1: RR
    \item Card 2: BB
    \item Card 3: RB
\end{itemize}

A card is selected at random and placed on the ground. If the upper side is red, find the probability that the other side is black.

\textbf{Solution:}  
Let
\begin{itemize}
    \item $RR, BB, RB$: events corresponding to the selected card
    \item $R$: event that the upturned side is red
\end{itemize}

Then,
\[
\Pr(RB \mid R) = \frac{\Pr(R \mid RB)\Pr(RB)}{\Pr(R \mid RR)\Pr(RR) + \Pr(R \mid RB)\Pr(RB) + \Pr(R \mid BB)\Pr(BB)}
\]
\[
= \frac{(1/2)(1/3)}{(1)(1/3) + (1/2)(1/3) + (0)(1/3)} = \frac{1}{3}
\]

\section{Random Variables}

\subsection{Motivation and Concept}

\textbf{Definition:}  
A \textit{random variable} is a real-valued function defined on the sample space of an experiment. Values are determined by outcomes of the experiment, and probabilities are assigned to these values.

\textbf{Visualization:}  
The distribution of a random variable can be represented as a bar diagram where:
\begin{itemize}
    \item The x-axis shows possible values
    \item The height at value $a$ is $\Pr[X = a]$
\end{itemize}

\subsection{Example — Tossing 3 Fair Coins}

\textbf{Problem Statement:}  
Let $Y$ denote the number of heads obtained when 3 fair coins are tossed.

\textbf{Solution:}  
Possible values: $Y = 0, 1, 2, 3$
\[
\Pr(Y = 0) = \Pr(t,t,t) = \frac{1}{8}
\]
\[
\Pr(Y = 1) = \Pr(t,t,h), (t,h,t), (h,t,t) = \frac{3}{8}
\]
\[
\Pr(Y = 2) = \Pr(t,h,h), (h,t,h), (h,h,t) = \frac{3}{8}
\]
\[
\Pr(Y = 3) = \Pr(h,h,h) = \frac{1}{8}
\]

Since $Y$ must take one of these values,
\[
\sum_{k=0}^3 \Pr(Y = k) = 1
\]

\section{Probability Mass Function (PMF)}

\subsection{Concept}

\textbf{Definition:}  
A random variable that can take at most a countable number of values is called \textit{discrete}.

Let $X$ be a discrete random variable with range
\[
R_X = \{x_1, x_2, x_3, \ldots\}
\]
The function
\[
p(x_k) = \Pr(X = x_k)
\]
is called the \textbf{Probability Mass Function (PMF)} of $X$.

\textbf{Property:}
\[
\sum_k p(x_k) = 1
\]
